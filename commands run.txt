    sudo apt-get install -y nodejs

    # 2. Install PM2 Globally
    sudo npm install pm2@latest -g

    # 3. Tell PM2 to start automatically if the server reboots
    pm2 startup systemd -u root --hp /root

    cat /opt/panel/www/config/database.php | grep DB_PASS

# We need to manually create that missing directory and empty file so Nginx can successfully reboot.
    # 1. Create the missing WAF directory
    sudo mkdir -p /etc/nginx/waf/

    # 2. Create the blank file that Nginx is crying about
    sudo touch /etc/nginx/waf/cp.uniqueballia.in.conf

# Delete the timezone line from the config
    sudo sed -i '/default-time-zone/d' /etc/mysql/mariadb.conf.d/50-server.cnf

# Restart the database
    sudo systemctl restart mariadb

# Now, we need to teach MariaDB all the named timezones in the world. Linux stores these in /usr/share/zoneinfo. We will pipe them directly into the MySQL system tables.
    sudo mysql_tzinfo_to_sql /usr/share/zoneinfo | sudo mysql mysql

# You asked how you can see exactly what the daemon is doing and why it's stuck. We can use a powerful Linux command called strace to look inside the brain of the Python process while it is running.
    sudo strace -p $(pgrep -f worker.py)

    # 2. Automatically grab the real DB password from your PHP config and inject it into Python
    REAL_PASS=$(grep "'DB_PASS'" /opt/panel/www/config/database.php | cut -d"'" -f4)
    sudo sed -i "s/YOUR_SECURE_PASSWORD/$REAL_PASS/g" /opt/panel/daemon/worker.py

    # 3. Secure the file, mark the stuck UI tasks as failed, and restart the daemon
    sudo chmod +x /opt/panel/daemon/worker.py
    mysql -e "UPDATE panel_core.tasks_queue SET status='failed', output_log='System reset to fix connection leak.' WHERE status='pending' OR status='processing';"
    sudo systemctl daemon-reload
    sudo systemctl restart panel-daemon
